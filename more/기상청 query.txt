< 기상청 DB 테이블 생성 (kmaweather) >
CREATE TABLE `kmaweather` (
	`no` INT(20) NOT NULL AUTO_INCREMENT,
	`area` VARCHAR(50) NULL DEFAULT NULL,
	`date` VARCHAR(255) NULL DEFAULT NULL,
	`day1_Mo` VARCHAR(255) NULL DEFAULT NULL,
	`day1_Af` VARCHAR(255) NULL DEFAULT NULL,
	`day2_Mo` VARCHAR(255) NULL DEFAULT NULL,
	`day2_Af` VARCHAR(255) NULL DEFAULT NULL,
	`day3_Mo` VARCHAR(255) NULL DEFAULT NULL,
	`day3_Af` VARCHAR(255) NULL DEFAULT NULL,
	`day4_Mo` VARCHAR(255) NULL DEFAULT NULL,
	`day4_Af` VARCHAR(255) NULL DEFAULT NULL,
	`day5_Mo` VARCHAR(255) NULL DEFAULT NULL,
	`day5_Af` VARCHAR(255) NULL DEFAULT NULL,
	`day6` VARCHAR(255) NULL DEFAULT NULL,
	`day7` VARCHAR(255) NULL DEFAULT NULL,
	`day8` VARCHAR(255) NULL DEFAULT NULL,
	PRIMARY KEY (`no`)
)
